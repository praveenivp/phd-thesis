\huge{Physical Constants}
\normalsize
\begin{itemize}
	\item Gyromagnetic ratio of hydrogen: $\gamma_H = 2\pi\times 42.577478461 \times10^{6}$ rad/Ts
	\item Gyromagnetic ratio of deuterium: $\gamma_D = 2\pi \times 6.536 \times10^{6}$ rad/Ts
	\item Boltzmann constant: $k_B = 1.380649 \times 10^{-23}$ J/K
	\item Planck constant: $h = 6.62607015 \times 10^{-34}$ Js
	\item Reduced Planck constant: $\hbar = \frac{h}{2\pi} = 1.0545718 \times 10^{-34}$ Js
\end{itemize}

\huge{Relevant Tissue Properties}
\normalsize
\begin{itemize}
	\item Natural abundance of deuterium: $\approx 0.0156\%$
	\item Concentration of deuterated water in the human brain: $\approx 10.12 mM\%$\cite{defeyterDeuteriumMetabolicImaging2018}
	\item Magnetic susceptibility of blood (oxygenated): $\chi_{blood,oxy} \approx -0.026$ ppm
	\item Magnetic susceptibility of blood (deoxygenated): $\chi_{blood,deoxy} \approx -0.034$ ppm
\end{itemize}
