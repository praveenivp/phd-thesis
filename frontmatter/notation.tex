\Large{Physical Constants}
\normalsize
\begin{itemize}
	\item Gyromagnetic ratio of hydrogen: $\gamma\textsubscript{$^1$H} \approx 2\pi\times 42.577478461 \times10^{6}$ rad/Ts
	\item Gyromagnetic ratio of deuterium: $\gamma\textsubscript{$^2$H} \approx 2\pi \times 6.536 \times10^{6}$ rad/Ts
	\item Boltzmann constant: $k_B \approx 1.380649 \times 10^{-23}$ J/K
	\item Planck constant: $h \approx 6.62607015 \times 10^{-34}$ Js
	\item Reduced Planck constant: $\hbar = \frac{h}{2\pi} \approx 1.0545718 \times 10^{-34}$ Js
\end{itemize}

\Large{Relevant Tissue Properties}
\normalsize
\begin{itemize}

	\item Magnetic susceptibility difference between oxygenated and deoxygenated blood: $\chi\textsubscript{deoxy}-\chi\textsubscript{oxy} \approx 0.18$ ppm~\cite{weisskoffMRISusceptometryImagebased1992}
	\item Fat-Water Chemical Shift: $\approx$ 3.3 ppm ~\cite{renCompositionAdiposeTissue2008}
		\item Natural abundance of deuterium: $\approx 0.0156\%$
	\item Concentration of deuterated water in the human brain: $\approx 10.12 mM\%$~\cite{defeyterDeuteriumMetabolicImaging2018}
\end{itemize}
